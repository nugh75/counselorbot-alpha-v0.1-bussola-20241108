# Comandi per la gestione del database FAISS (backend.py su localhost:8000)

# Caricare un file
curl -X POST "http://localhost:8000/faiss/upload" -F "file=@/path/to/file.pdf"

# Elencare documenti
curl -X GET "http://localhost:8000/faiss/list"

# Rimuovere un file (filename esatto)
curl -X DELETE "http://localhost:8000/faiss/remove-file/nomefile.pdf"
# Con spazi: curl -X DELETE "http://localhost:8000/faiss/remove-file/Imparare%20a%20dirigere%20se%20stessi.pdf"
